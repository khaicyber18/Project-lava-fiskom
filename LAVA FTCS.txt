    <script>
        // ==========================================
        // 1. GENERASI TOPOGRAFI 3D MOCKUP (Javascript)
        // ==========================================
        const N = 50; // Resolusi disesuaikan untuk browser agar ringan
        const L = 40.0;
        const x = [], y = [], z = [];

        for (let i = 0; i < N; i++) {
            x.push((i / (N - 1)) * L);
            y.push((i / (N - 1)) * L);
            z[i] = [];
        }

        const x_kawah = 20.0;
        const y_kawah = 32.0;

        // Bikin Gunung mirip di Python
        for (let i = 0; i < N; i++) {
            for (let j = 0; j < N; j++) {
                const X = x[j];
                const Y = y[i];
                const R2 = Math.pow(X - x_kawah, 2) + Math.pow(Y - y_kawah, 2);
                
                // Rumus Gaussian untuk bentuk gunung
                let Z = 850 * Math.exp(-R2 / (2 * Math.pow(8.0, 2)));
                
                // Tambahan elevasi kasar
                Z = Math.max(Z, 10); 
                z[i][j] = Z;
            }
        }

        // ==========================================
        // 2. RENDER KE PLOTLY
        // ==========================================
        const data = [{
            z: z,
            x: x,
            y: y,
            type: 'surface',
            colorscale: 'Earth', // Mirip warna vegetasi/gelap
            showscale: false
        }];

        const layout = {
            paper_bgcolor: '#111827',
            plot_bgcolor: '#111827',
            margin: { t: 0, b: 0, l: 0, r: 0 },
            scene: {
                xaxis: { visible: false },
                yaxis: { visible: false },
                zaxis: { visible: false },
                camera: {
                    eye: { x: 1.5, y: -1.5, z: 1.2 } // Mirip elev=45, azim=55
                }
            }
        };

        Plotly.newPlot('plot3d', data, layout, {responsive: true});

        // ==========================================
        // 3. SIMULASI DATA EWS (Animasi UI Angka)
        // ==========================================
        let waktu = 1.1;
        let jarak = 2.36;

        setInterval(() => {
            waktu += 0.1;
            // Simulasi lava makin turun/mendekat
            if (jarak > 0.1) jarak -= 0.02; 
            
            // Format string desimal yang rapi dan konsisten
            document.getElementById('waktu-sim').innerText = waktu.toFixed(1);
            document.getElementById('jarak-lava').innerText = Math.max(0, jarak).toFixed(2);
            
            // Ubah status dan warna jika lava makin dekat
            const statusEl = document.getElementById('status-ews');
            if (jarak <= 0.1) {
                statusEl.innerText = "EVAKUASI!";
                statusEl.style.color = "#FF0000";
            } else if (jarak <= 1.0) {
                statusEl.innerText = "AWAS";
                statusEl.style.color = "#FF3300";
            }
        }, 1000); // Update setiap 1 detik
    </script>
